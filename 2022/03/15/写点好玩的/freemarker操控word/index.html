import freemarker.template.Configuration;
import freemarker.template.Template;
import sun.misc.BASE64Encoder;

import java.util.Map;

public class GenWord {

    /**
     * з”ҹжҲҗwordж–Үд»¶
     *
     * @param dataMap      ж•°жҚ®
     * @param templateName жЁЎжқҝеҗҚз§°
     * @param file         ж–Үд»¶
     * @param fileName     ж–Үд»¶еҗҚз§°
     */
    public static void createWord(Map<String, Object> dataMap, String templateName, File file, String fileName) {
        try {
            Configuration configuration = new Configuration();
            configuration.setDefaultEncoding("UTF-8");
            configuration.setDirectoryForTemplateLoading(file);
            Template template = configuration.getTemplate(templateName);
            File outFile = new File("E:\\87b0767d5d8f49039fac2b1d1e29fdc3\\" + fileName);
            if (!outFile.getParentFile().exists()) {
                outFile.getParentFile().mkdirs();
            }
            Writer out = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(outFile), "UTF-8"));
            template.process(dataMap, out);
            out.flush();
            out.close();
        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    public static String convertFileToBase64(String imgPath) throws IOException {
        InputStream in = new FileInputStream(imgPath);
        ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
        // е°ҶеҶ…е®№иҜ»еҸ–еҶ…еӯҳдёӯ
        byte[] buffer = null;
        buffer = new byte[1024];
        int len = -1;
        while ((len = in.read(buffer)) != -1) {
            outputStream.write(buffer, 0, len);
        }
        buffer = outputStream.toByteArray();

        if (in != null) {
            try {
                // е…ій—ӯinputStreamжөҒ
                in.close();
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
        if (outputStream != null) {
            try {
                // е…ій—ӯoutputStreamжөҒ
                outputStream.close();
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
            // еҜ№еӯ—иҠӮж•°з»„Base64зј–з Ғ
            return new BASE64Encoder().encode(buffer);

    }
}
