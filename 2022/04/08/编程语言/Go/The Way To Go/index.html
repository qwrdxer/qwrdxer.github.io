    еҸҚе°„жҳҜз”ЁзЁӢеәҸжЈҖжҹҘе…¶жүҖжӢҘжңүзҡ„з»“жһ„пјҢе°Өе…¶жҳҜзұ»еһӢзҡ„дёҖз§ҚиғҪеҠӣпјӣиҝҷжҳҜе…ғзј–зЁӢзҡ„дёҖз§ҚеҪўејҸгҖӮеҸҚе°„еҸҜд»ҘеңЁиҝҗиЎҢж—¶жЈҖжҹҘзұ»еһӢе’ҢеҸҳйҮҸпјҢдҫӢеҰӮе®ғзҡ„еӨ§е°ҸгҖҒж–№жі•е’Ң еҠЁжҖҒ зҡ„и°ғз”Ёиҝҷдәӣж–№жі•гҖӮиҝҷеҜ№дәҺжІЎжңүжәҗд»Јз Ғзҡ„еҢ…е°Өе…¶жңүз”ЁгҖӮиҝҷжҳҜдёҖдёӘејәеӨ§зҡ„е·Ҙе…·пјҢйҷӨйқһзңҹеҫ—жңүеҝ…иҰҒпјҢеҗҰеҲҷеә”еҪ“йҒҝе…ҚдҪҝз”ЁжҲ–е°ҸеҝғдҪҝз”ЁгҖӮ

    еҸҳйҮҸзҡ„жңҖеҹәжң¬дҝЎжҒҜе°ұжҳҜзұ»еһӢе’ҢеҖјпјҡеҸҚе°„еҢ…зҡ„ Type з”ЁжқҘиЎЁзӨәдёҖдёӘ Go зұ»еһӢпјҢеҸҚе°„еҢ…зҡ„ Value дёә Go еҖјжҸҗдҫӣдәҶеҸҚе°„жҺҘеҸЈгҖӮ

    Value жңүдёҖдёӘ Type ж–№жі•иҝ”еӣһ reflect.Value зҡ„ TypeгҖӮеҸҰдёҖдёӘжҳҜ Type е’Ң Value йғҪжңү Kind ж–№жі•иҝ”еӣһдёҖдёӘеёёйҮҸжқҘиЎЁзӨәзұ»еһӢпјҡUintгҖҒFloat64гҖҒSlice зӯүзӯүгҖӮеҗҢж · Value жңүеҸ«еҒҡ Int е’Ң Float зҡ„ж–№жі•еҸҜд»ҘиҺ·еҸ–еӯҳеӮЁеңЁеҶ…йғЁзҡ„еҖјпјҲи·ҹ int64 е’Ң float64 дёҖж ·пјү

    йҖҡиҝҮеҸҚе°„дҝ®ж”№еҖј

    еҪ“ v := reflect.ValueOf(x) еҮҪж•°йҖҡиҝҮдј йҖ’дёҖдёӘ x жӢ·иҙқеҲӣе»әдәҶ vпјҢйӮЈд№Ҳ v зҡ„ж”№еҸҳе№¶дёҚиғҪжӣҙж”№еҺҹе§Ӣзҡ„ xгҖӮиҰҒжғі v зҡ„жӣҙж”№иғҪдҪңз”ЁеҲ° xпјҢйӮЈе°ұеҝ…йЎ»дј йҖ’ x зҡ„ең°еқҖ v = reflect.ValueOf(&x)гҖӮ

    еҪ“з„¶пјҢжӯӨж—¶зҡ„vд»Қж— жі•иҝӣиЎҢдҝ®ж”№пјҢйңҖи°ғз”Ёv.Elem()еҮҪж•°пјҢ

    еҸҚе°„з»“жһ„дҪ“

    жңүдәӣж—¶еҖҷйңҖиҰҒеҸҚе°„дёҖдёӘз»“жһ„зұ»еһӢгҖӮNumField() ж–№жі•иҝ”еӣһз»“жһ„еҶ…зҡ„еӯ—ж®өж•°йҮҸпјӣйҖҡиҝҮдёҖдёӘ for еҫӘзҺҜз”Ёзҙўеј•еҸ–еҫ—жҜҸдёӘеӯ—ж®өзҡ„еҖј Field(i)гҖӮ

    жҲ‘д»¬еҗҢж ·иғҪеӨҹи°ғз”ЁзӯҫеҗҚеңЁз»“жһ„дёҠзҡ„ж–№жі•пјҢдҫӢеҰӮпјҢдҪҝз”Ёзҙўеј• n жқҘи°ғз”ЁпјҡMethod(n).Call(nil)гҖӮ

    еҗ‘еҸҚе°„ж–№жі•дёӯдј е…ҘеҸӮж•°

    и®ҫзҪ®з»“жһ„дҪ“жҲҗе‘ҳзҡ„еҖј

    пјҲжҲҗе‘ҳзҡ„йҰ–еӯ—жҜҚеә”иҜҘеӨ§еҶҷпјҢеҗҰеҲҷж— жі•и®ҝй—®пјү

    жҺҘеҸЈе’ҢеҠЁжҖҒзұ»еһӢ

    жҺҘж”¶дёҖдёӘпјҲжҲ–еӨҡдёӘпјүжҺҘеҸЈзұ»еһӢдҪңдёәеҸӮж•°зҡ„еҮҪж•°пјҢе…¶е®һеҸӮеҸҜд»ҘжҳҜд»»дҪ•е®һзҺ°дәҶиҜҘжҺҘеҸЈзҡ„зұ»еһӢгҖӮ е®һзҺ°дәҶжҹҗдёӘжҺҘеҸЈзҡ„зұ»еһӢеҸҜд»Ҙиў«дј з»ҷд»»дҪ•д»ҘжӯӨжҺҘеҸЈдёәеҸӮж•°зҡ„еҮҪж•° гҖӮ

    еҠЁжҖҒж–№жі•и°ғз”Ё

    й”ҷиҜҜеӨ„зҗҶ

    Go жЈҖжҹҘе’ҢжҠҘе‘Ҡй”ҷиҜҜжқЎд»¶зҡ„жғҜжңүж–№ејҸпјҡ

    • дә§з”ҹй”ҷиҜҜзҡ„еҮҪж•°дјҡиҝ”еӣһдёӨдёӘеҸҳйҮҸпјҢдёҖдёӘеҖје’ҢдёҖдёӘй”ҷиҜҜз ҒпјӣеҰӮжһңеҗҺиҖ…жҳҜ nil е°ұжҳҜжҲҗеҠҹпјҢйқһ nil е°ұжҳҜеҸ‘з”ҹдәҶй”ҷиҜҜгҖӮ
    • дёәдәҶйҳІжӯўеҸ‘з”ҹй”ҷиҜҜж—¶жӯЈеңЁжү§иЎҢзҡ„еҮҪж•°пјҲеҰӮжһңжңүеҝ…иҰҒзҡ„иҜқз”ҡиҮідјҡжҳҜж•ҙдёӘзЁӢеәҸпјүиў«дёӯжӯўпјҢеңЁи°ғз”ЁеҮҪж•°еҗҺеҝ…йЎ»жЈҖжҹҘй”ҷиҜҜгҖӮ

    дёӢйқўиҝҷж®өжқҘиҮӘ pack1 еҢ…зҡ„д»Јз Ғ Func1 жөӢиҜ•дәҶе®ғзҡ„иҝ”еӣһеҖјпјҡ

    д»Һ panic дёӯжҒўеӨҚпјҲRecoverпјү

    жӯЈеҰӮеҗҚеӯ—дёҖж ·пјҢиҝҷдёӘпјҲrecoverпјүеҶ…е»әеҮҪж•°иў«з”ЁдәҺд»Һ panic жҲ– й”ҷиҜҜеңәжҷҜдёӯжҒўеӨҚпјҡи®©зЁӢеәҸеҸҜд»Ҙд»Һ panicking йҮҚж–°иҺ·еҫ—жҺ§еҲ¶жқғпјҢеҒңжӯўз»ҲжӯўиҝҮзЁӢиҝӣиҖҢжҒўеӨҚжӯЈеёёжү§иЎҢгҖӮ

    recover еҸӘиғҪеңЁ defer дҝ®йҘ°зҡ„еҮҪж•°пјҲеҸӮи§Ғ 6.4 иҠӮпјүдёӯдҪҝз”Ёпјҡз”ЁдәҺеҸ–еҫ— panic и°ғз”Ёдёӯдј йҖ’иҝҮжқҘзҡ„й”ҷиҜҜеҖјпјҢеҰӮжһңжҳҜжӯЈеёёжү§иЎҢпјҢи°ғз”Ё recover дјҡиҝ”еӣһ nilпјҢдё”жІЎжңүе…¶е®ғж•ҲжһңгҖӮ

    жҖ»з»“пјҡpanic дјҡеҜјиҮҙж Ҳиў«еұ•ејҖзӣҙеҲ° defer дҝ®йҘ°зҡ„ recover() иў«и°ғз”ЁжҲ–иҖ…зЁӢеәҸдёӯжӯўгҖӮ

    дёӢйқўдҫӢеӯҗдёӯзҡ„ protect еҮҪж•°и°ғз”ЁеҮҪж•°еҸӮж•° g жқҘдҝқжҠӨи°ғз”ЁиҖ…йҳІжӯўд»Һ g дёӯжҠӣеҮәзҡ„иҝҗиЎҢж—¶ panicпјҢе№¶еұ•зӨә panic дёӯзҡ„дҝЎжҒҜпјҡ
